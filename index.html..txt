        // Variables pour mémoriser l'état actuel des filtres
        let currentSearchQuery = "";

        // JAVASCRIPT - Le Moteur d'Affichage et de Filtrage
        function render() {
            const list = document.getElementById('events-list');
            
            // 1. Filtrer les événements selon la recherche ET l'onglet
            let filteredEvents = events.filter(e => {
                // Vérifier la recherche (nom ou ville)
                const textToSearch = (e.name + " " + e.city).toLowerCase();
                const matchesSearch = textToSearch.includes(currentSearchQuery.toLowerCase());
                
                // Vérifier l'onglet
                let matchesTab = true;
                if (currentTabFilter === "FR") matchesTab = e.country === "FR";
                if (currentTabFilter === "ES") matchesTab = e.country === "ES";
                if (currentTabFilter === "basque") matchesTab = e.region === "basque";

                return matchesSearch && matchesTab;
            });

            // 2. Trier par date
            filteredEvents.sort((a, b) => new Date(a.date) - new Date(b.date));

            // 3. Afficher un message si aucun résultat
            if (filteredEvents.length === 0) {
                list.innerHTML = `<div class="no-results">Aucune fête ne correspond à votre recherche 🐂</div>`;
                return;
            }

            // 4. Créer le HTML pour les événements filtrés
            list.innerHTML = filteredEvents.map(e => {
                const eventDate = new Date(e.date);
                eventDate.setHours(0,0,0,0);
                const today = new Date();
                today.setHours(0,0,0,0);
                
                const diffTime = eventDate - today;
                const diffDays = Math.ceil(diffTime / (1000 * 60 * 60 * 24));
                
                let countdownText = diffDays < 0 ? "L'événement est passé" : diffDays === 0 ? "C'est aujourd'hui !" : `Dans ${diffDays} jours`;
                const flag = e.country === "FR" ? "🇫🇷" : "🇪🇸";
                
                return `
                    <div class="card ${e.country.toLowerCase()} ${diffDays < 0 ? 'past' : ''}">
                        <div class="card-header">
                            <div>
                                <h2>${flag} ${e.name}</h2>
                                <div class="location">${e.city}</div>
                            </div>
                        </div>
                        <span class="badge">👥 ${e.visitors} visiteurs</span>
                        <div class="date-info">📅 ${eventDate.toLocaleDateString('fr-FR', {day: 'numeric', month: 'long', year: 'numeric'})}</div>
                        <div class="description">${e.desc}</div>
                        <div class="countdown">${countdownText}</div>
                        <a href="${e.link}" target="_blank" class="btn">Visiter le site officiel</a>
                    </div>
                `;
            }).join('');
